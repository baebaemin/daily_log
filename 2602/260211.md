### [React] react-hook-form + zod

- react-hook-form 기본 사용법

  ```jsx
  import { useForm } from "react-hook-form";

  function SimpleForm() {
    const {
      register,
      handleSubmit,
      formState: { errors },
    } = useForm();

    const onSubmit = (data) => {
      console.log(data); // { email: "...", password: "..." }
    };

    return (
      <form onSubmit={handleSubmit(onSubmit)}>
        <input
          {...register("email", {
            required: "이메일을 입력해주세요",
            pattern: {
              value: /^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}$/i,
              message: "올바른 이메일 형식이 아닙니다",
            },
          })}
          placeholder="이메일"
        />
        {errors.email && <span>{errors.email.message}</span>}

        <input
          type="password"
          {...register("password", {
            required: "비밀번호를 입력해주세요",
            minLength: { value: 8, message: "8자 이상 입력해주세요" },
          })}
          placeholder="비밀번호"
        />
        {errors.password && <span>{errors.password.message}</span>}

        <button type="submit">로그인</button>
      </form>
    );
  }
  ```

- react-hook-form + zod + TypeScript 조합

  ```tsx
  import { useForm } from "react-hook-form";
  import { zodResolver } from "@hookform/resolvers/zod";
  import { z } from "zod";

  const schema = z.object({
    email: z.string().email("올바른 이메일을 입력해주세요"),
    password: z.string().min(8, "8자 이상 입력해주세요"),
  });

  type FormData = z.infer<typeof schema>;

  function FormWithZod() {
    const {
      register,
      handleSubmit,
      formState: { errors },
    } = useForm<FormData>({
      resolver: zodResolver(schema),
    });

    const onSubmit = (data: FormData) => {
      console.log(data);
    };

    return (
      <form onSubmit={handleSubmit(onSubmit)}>
        <input {...register("email")} />
        {errors.email && <span>{errors.email.message}</span>}

        <input type="password" {...register("password")} />
        {errors.password && <span>{errors.password.message}</span>}

        <button type="submit">제출</button>
      </form>
    );
  }
  ```

  // email 값 설정

### [React] react-hook-form > trigger

- 기본 사용법

```jsx
const {
  register,
  trigger,
  formState: { erros },
} = useForm();

// 특정 필드 validation 실행
await trigger();

// 여러 필드 동시에 validation
await trigger(["email", "password"]);

// 모든 필드 validation
await trigger();

// NOTE 1: trigger는 validation 로직이 아래와 같이 비동기일 수 있기 때문에 Promise를 반환한다
register("username", {
  validate: async (value) => {
    // 서버에 중복 확인 요청
    const response = await fetch("/api/check-username?name=${value}");
    const data = await response.json();
    return data.available || "이미 사용 중인 아이디입니다";
  },
});

// NOTE 2: validation 로직이 아래와 같이 동기여도 일관성있는 API를 위해 Promise를 반환한다
register("email", {
  required: "필수입니다",
  minLength: { value: 5, message: "5자 이상" },
});
```

  2. 다음 단계로 넘어가기 전 검증

  ```jsx
  function MultiStepForm() {
    const { register, trigger } = useForm();
    const [step, setStep] = useState(1);

    const goToNextStep = async () => {
      // step 1의 필드들만 검증
      const isValid = await trigger(["name", "email"]);

      if (isValid) {
        setStep(2);
      }
    };

    return (
      <>
        {step === 1 && (
          <>
            <input {...register("name", { required: true })} />
            <input {...register("email", { required: true })} />
            <button type="button" onClick={goToNextStep}>
              다음
            </button>
          </>
        )}
        {step === 2 && (
          <>
            <input {...register("address")} />
            <button type="submit">제출</button>
          </>
        )}
      </>
    );
  }
  ```

  3. 조건부 validation

  ```jsx
  function ConditionalForm() {
    const { register, trigger, watch, getValues } = useForm();
    const agreeToTerms = watch("agreeToTerms");

    const handleCheckboxChange = async () => {
      const currentValue = getValues("agreeToTerms"); // 현재 체크 상태

      // 체크박스 선택시 다른 필드 재검증
      if (currentValue) {
        await trigger(["email", "password"]);
      }
    };

    return (
      <>
        <input {...register("email")} />
        <input {...register("password")} />
        <input
          type="checkbox"
          {...register("agreeToTerms")}
          onChange={handleCheckboxChange}
        />
      </>
    );
  }
  ```
