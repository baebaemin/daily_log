function Radio({

  isChecked,
  readOnly,
  ...remainProps
}: RadioProps) {
  const handleRadioChange = (event: React.ChangeEvent<HTMLInputElement>) => {

      id={radioId}

      onChange={handleRadioChange}
      checked={isChecked}

    >
      <label htmlFor={radioId}>{children}</label>
    </input>
  );
}
