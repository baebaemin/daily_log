### [TypeScript] 조건부 렌더링시 타입 가드

```tsx
function ServiceModal({ isShow, serviceInfo }: ServiceModalProps) {
  if (isShow || serviceInfo === null) {
    return children;
  }

  // (property) DOMAttributes<HTMLSpanElement>.dangerouslySetInnerHTML?: {
  //    __html: string | TrustedHTML;
  // } | undefined

  return <span dangerouslySetInnerHTML={{ __html: serviceInfo!.title || "" }} />;
}
```

- serviceInfo가 null이면 early return을 하고 있는데도 dangerouslySetInnerHTML에 전달하는 값에 Non-null Assertion 연산자 `!`를 사용해야 한다.

  - `serviceInfo.title`을 전달하면 `TS18048: 'serviceInfo' is possibly 'undefined'.` Type Error 발생

- 조건을 분리해야 serviceInfo가 non-null로 추론됨

```tsx
function ServiceModal({ isShow, serviceInfo }: ServiceModalProps) {
  if (isShow) {
    return children;
  }

  if (serviceInfo === null) {
    return children;
  }

  return <span dangerouslySetInnerHTML={{ __html: serviceInfo.title || "" }} />;
}
```
